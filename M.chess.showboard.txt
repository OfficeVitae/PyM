; displays the #board fields, showing the pieces and whether or not it is a check piece
; appends a + to the king of the current player if currently in check
; and a * to the piece if it checks the other king
; to the left we show where the current player can check the opponent (!) or whether it checks the opponent(+)
; pass in a number of positions in #showpositions to mark with the index in the right-top corner of a field in the presented color in #showcolor
;;out("Show positions: ",showpositions,lf) ; debugging
(out('   '),for(col,1..7,out(cols(col),'    ')),out(cols(8),lf))
for(row,ranges(playerindex)
   ,out(' ')
   ,for(col,1..8
       ,showpositionindex=len(showpositions),while(and(showpositionindex>0,or(showpositions(showpositionindex,2)!=col,showpositions(showpositionindex,1)!=row)),showpositionindex-=1)
       ,bgindex=1+((row%2)^(col%2)),bgc=bgcolors(bgindex),
       ,outc((shadowcolors(bgindex),bgc),'  ',cols(col))
       ,if(and(showpositionindex>0,len(showpositions(showpositionindex))>4,showpositions(showpositionindex,5)<=0)
          ,outc((showcolor,bgc),showpositions(showpositionindex,4))
          ,outc((shadowcolors(bgindex),bgc),' ')
          )
       ,if(and(showpositionindex>0,or(len(showpositions(showpositionindex))<=4,showpositions(showpositionindex,5)>0))
          ,outc((if(len(showpositions(showpositionindex))>4,movetypecolors(showpositions(showpositionindex,5)),showcolor),bgc),showpositions(showpositionindex,4))
          ,outc((showcolor,bgc),' ')
          )
       )
   ,out(lf)
   ,out(row)
   ,for(col,1..8
       ,showpositionindex=len(showpositions),while(and(showpositionindex>0,or(showpositions(showpositionindex,2)!=col,showpositions(showpositionindex,1)!=row)),showpositionindex-=1)
       ,positionmoveeffects=if(showpositionindex>0,showpositions(showpositionindex,3),0),draweffect=(positionmoveeffects&1),incheckeffect=(positionmoveeffects>>1)
       ,p=board(row,col),pl=(3-sign(p))\2,bgindex=1+((row%2)^(col%2)),bg=bgcolors(bgindex)
       ,outc((shadowcolors(bgindex),bg),rows(row))
       ,if(abs(p)==6&&pl==playerindex&&moveeffect>0,outc(9,bg),moveeffectcharacters(moveeffect)),outc((colors(playerindex),bg),getCheckPieceCharacter(3-playerindex,row,col)))
       ,outc((colors(pl),bg),if(p!=0,piecetypes(abs(p)),' '))
       ,if(incheckeffect>0,outc((colors(playerindex),bg),moveeffectcharacters(2*incheckeffect-1)),outc((colors(3-playerindex),bg),getCheckPieceCharacter(playerindex,row,col)))
       ,if(draweffect>0,outc((colors(playerindex),bg),'='),outc((shadowcolors(bgindex),bg),rows(row)))
       )
   ,out(row,lf)
   ,out(' ')
   ,for(col,1..8,bgindex=1+((row%2)^(col%2)),outc((shadowcolors(bgindex),bgcolors(bgindex)),'  ',cols(col),'  '))
   ,out(lf)
  )
(out('   '),for(col,1..7,out(cols(col),'    ')),out(cols(8),lf))
